#version 330

in vec4 vPosition;
in vec4 cColor;
out vec4 outColor;

uniform float time = 0.0f;

void main(){
	outColor = vPosition;
	vec4 cPos = vPosition;
	outColor.r = (cos(time*1.3f + cPos.x)+1.0f)/2.0f;
	outColor.g = (sin(time*1.3f + cPos.y)+1.0f)/2.0f;
	outColor.b = vPosition.x;

	float rad = sqrt(cPos.x * cPos.x + cPos.y * cPos.y);
	if( rad < (cos(time) + 2.0f)/2.0f && rad > (cos(time)+2.0f)/4.0f){
		outColor = vec4(1.0f, 1.0f, 1.0f, 1.0f) - sin(time*1.3f + cPos.y)+1.0f)/2.0f;
	}
}
