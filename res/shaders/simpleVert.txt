#version 330

layout(location = 0) in vec4 position;
layout(location = 1) in vec4 color;

out vec4 vColor;
out vec4 vPosition;

layout(location = 3) uniform float time = 0.0f;

void main(){
	vColor = color;
	vPosition = position;
	gl_Position = position;

	gl_Position.y = cos(time + position.x) / 2.0f;
}
